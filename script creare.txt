set serveroutput on;

drop table users;
drop table products;
drop table categories;
drop sequence userID_seq;
drop sequence catID_seq;

create table users (
userID number not null,
pw number not null,
firstName varchar2(20),
lastName varchar2(20),
adresa varchar2(20),
privilegies varchar2(20),
phoneNumber number,
primary KEY(userID)
)



create table categories(
catName varchar2(20),
catId number not null,
primary KEY(catId)
)

create table products(
prodID number not null,
prodName varchar(30),
prodDesc varchar(250),
prodPrice float,
primary key(prodID),
catId number not null,
initialStock number,
currentStock number,
constraint catId foreign KEY (catId) references categories(catId)
)







Create sequence userID_seq
minvalue 1
start with 1
increment by 1
cache 20;



Create sequence catID_seq
minvalue 1
start with 1
increment by 1
cache 20;

create sequence prodID_seq
minvalue 1
start with 1
increment by 1
cache 20;
